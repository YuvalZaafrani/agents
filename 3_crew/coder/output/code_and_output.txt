```python
# Calculation of the series: 1 - 1/3 + 1/5 - 1/7 + ... for the first 10,000 terms
def calculate_series(n_terms):
    total = 0.0
    for i in range(n_terms):
        term = 1 / (2 * i + 1)  # Odd numbers: 1, 3, 5, 7, ...
        if i % 2 == 0:  # even index: add
            total += term
        else:  # odd index: subtract
            total -= term
    return total * 4  # multiply the final result by 4

# Calculate the series for the first 10,000 terms
result = calculate_series(10000)

# Print the result
print("The result of the series multiplied by 4 is:", result)

# Output of the code
# The result of the series multiplied by 4 is: 3.1414926535900345
```
